可展开收缩的View

发表于 2018-04-24

ExpansionLayout


描述
点击自动收起展开的自定义布局
思路



  1. 继承LinearLayout，左边添加一个TextView（宽度权重为1），右边是一个图片（为了美化）


 private void initView(Context context) {
        textView = new TextView(context);
        textView.setLayoutParams(new LayoutParams(0, LinearLayout.LayoutParams.WRAP_CONTENT, 1));
        textView.setTextSize(TypedValue.COMPLEX_UNIT_PX, textSize);
        textView.setTextColor(textColor);
        setText(text);
        textView.setEllipsize(TextUtils.TruncateAt.END);
        textView.setOnClickListener(this);
        this.addView(textView, 0);
        imageView = new ImageView(context);
        imageView.setLayoutParams(new LayoutParams(LinearLayout.LayoutParams.WRAP_CONTENT, LinearLayout.LayoutParams.WRAP_CONTENT));
        imageView.setImageResource(expansionIcon);
        imageView.setScaleType(ImageView.ScaleType.CENTER_INSIDE);
        imageView.setPadding(DensityUtils.dip2px(context, space), 0, 0, 0);
        imageView.setOnClickListener(this);
        this.addView(imageView, 1);
    }


  1. 自定义属性


 <declare-styleable name=“ExpansionLayout”>
        <attr name=“expansionIcon” format=“reference” />//图片id
        <attr name=“expansionTextColor” format=“color|reference” />//字体颜色
        <attr name=“expansionTextSize” format=“dimension” />//字体大小
        <attr name=“space” format=“dimension” />//文字与图片的间距
        <attr name=“lines” format=“integer” />//收起时的行数
        <attr name=“expansionText” format=“string|reference” />//文本
    </declare-styleable>


  1. 暴露设置文本的方法，设置文本时测量展开和收起的高度


public void setText(String text) {
        this.text = text;
        textView.setMaxLines(Integer.MAX_VALUE);
        textView.setText(text);
        Rect rect = new Rect();
        int lineCount = textView.getLineCount();
        textView.getLineBounds(1, rect);
        openedHeight = Math.abs(rect.height()  lineCount) + paddingTop + paddingBottom;
        if (lineCount >= lines) {
            if (expansioned) {
                textView.setMaxLines(Integer.MAX_VALUE);
            } else {
                textView.setMaxLines(lines);
            }
            closedHeight = Math.abs(rect.height()  lines) + paddingTop + paddingBottom;
        } else {
            closedHeight = openedHeight;
        }
    }


  1. 设置点击事件


@Override
    public void onClick(View v) {
        if (expansioned) {
            imageView.animate().rotation(-180).setDuration(300).start();
            if (openedHeight != closedHeight) {
                anim(openedHeight, closedHeight);
            }
        } else {
            imageView.animate().rotation(0).setDuration(300).start();
            if (openedHeight != closedHeight) {
                anim(closedHeight, openedHeight);
            }
        }
        expansioned = !expansioned;
    }


  1. 展开收起动画效果


private void anim(int startValue, int endValue) {
        ValueAnimator valueAnimator = ValueAnimator
                .ofInt(startValue, endValue)
                .setDuration(300);
        valueAnimator
                .setInterpolator(new LinearInterpolator());
        valueAnimator.addListener(new AnimatorListenerAdapter() {
            @Override
            public void onAnimationStart(Animator animation) {
                super.onAnimationStart(animation);
                if (!expansioned) {
                    textView.setMaxLines(Integer.MAX_VALUE);
                }
            }
            @Override
            public void onAnimationEnd(Animator animation) {
                super.onAnimationEnd(animation);
                if (!expansioned) {
                    int lineCount = textView.getLineCount();
                    if (lineCount >= lines) {
                        textView.setMaxLines(lines);
                    }
                }
            }
        });
        valueAnimator.addUpdateListener(new ValueAnimator.AnimatorUpdateListener() {
            @Override
            public void onAnimationUpdate(ValueAnimator animation) {
                LinearLayout.LayoutParams layoutParams = (LinearLayout.LayoutParams) getLayoutParams();
                layoutParams.height = ((int) animation.getAnimatedValue());
                setLayoutParams(layoutParams);
            }
        });
        valueAnimator.start();
    }
